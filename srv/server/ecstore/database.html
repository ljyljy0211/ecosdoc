Database相关配置

mysql 配置

#数据库用户名
define('DB_USER', '');

# 数据库密码
define('DB_PASSWORD', ''); 

# 数据库名
define('DB_NAME', '');

#数据库地址
define('DB_HOST', ''); 

#是否启用长连接
define('DB_PCONNECT',1); 

#表名前缀
define('DB_PREFIX', 'sdb_'); 

#kvstore存储数据时，是否另存一份mysql里，false为开启，true为关闭。
#开启会影响性能。不过在key-value丢失情况下，可以使用命令行帮助重建。
#如果key-value引擎是高可用集群的话，建议关闭。
define('WITHOUT_KVSTORE_PERSISTENT', false); 

#字符集设置，建议使用utf8
define('DB_CHARSET', 'utf8');
define('DB_COLLATE', '');

#数据库集群,从库的连接参数
#当查询中没有使用事务时，将从slave中读取数据，
#实现Master-Slave，主从读写分离。
define('DB_SLAVE_NAME',DB_NAME);
define('DB_SLAVE_USER',DB_USER);
define('DB_SLAVE_PASSWORD',DB_PASSWORD);
define('DB_SLAVE_HOST',DB_HOST);
