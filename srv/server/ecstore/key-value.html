Key-Value引擎相关配置

Key-Value 存储引擎

#选项有：
# base_kvstore_filesystem   文件存储（默认）
# base_kvstore_mysql  mysql
# base_kvstore_memcache  memcache
# base_kvstore_dba  dba
# base_kvstore_tokyotyrant  tokyotyrant
# base_kvstore_flare  flare
# base_kvstore_mongodb  mongodb
# base_kvstore_tair  tair
define('KVSTORE_STORAGE', '');

#当使用kv存储引擎支持并使用memcache协议时，如：memcahe,tt,flare,需配置以下选项
#参见cache相关配置中的CACHE_MEMCACHE_CONFIG选项
define('KVSTORE_MEMCACHE_CONFIG', '');

#如使用mongodb，需配置以下选项
#mongodb 服务器配置
#连接服务器地址:
#格式如下："mongodb://${username}:${password}@localhost" , "mongodb:///tmp/mongo-27017.sock"
#define('MONGODB_SERVER_CONFIG', 'mongodb://localhost:27017');
#连接参数:
#array("connect" => TRUE),array("username"=>'xxxx', "password"=>'xxx');
#define('MONGODB_OPTION_CONFIG','return '. var_export(array('connect'=>true),1).';');

# KV_PREFIX KV引擎前缀
# 默认是default
# define('KV_PREFIX', 'default');

# 是否在写入kv时，先在mysql里存一份
# true为不开启，false为开启(默认)
# 开启时会小量影响性能，但支持kv重建
# 如果做了kv服务器高可用，建议关闭该选项
define('WITHOUT_KVSTORE_PERSISTENT', false);
