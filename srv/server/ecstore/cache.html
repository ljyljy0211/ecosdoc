Cache 相关配置

Cache 相关配置

#cache控制选项，true为开启，false关闭
#cache为临时性缓存，调试时可以关闭；正常使用时开启，可以提高性能
define('WITHOUT_CACHE',false);

#定义cache引擎；选项有：
#base_cache_nocache 不使用缓存
#base_cache_secache 使用文件缓存，该项为默认选项
#base_cache_memcache 使用memcache ** 推荐使用 **
#base_cache_memcached 使用memcached
#base_cache_apc 使用apc做为缓存引擎
#base_cache_eaccelerator 使用eaccelerator做为缓存引擎
#base_cache_xcache 使用xcache做为缓存引擎
#base_cache_wincache 使用Windows Cache做为缓存引擎
define('CACHE_STORAGE', '');

#如果使用memcache做为缓存引擎，则需提供memcache的相关连接参数
#可选如下几种方式
# 1.通过memcached服务端socket文件访问，如：'unix:///tmp/memcached.sock'
# 2.直接通过主机名或IP，以及memcache端口号（默认是11211）访问，如： '127.0.0.1:11211'
# 如有多个memcache服务，可以使用逗号隔开，如：'unix:///tmp/memcached.sock,127.0.0.1:11211,127.0.0.1:11212'
define('CACHE_MEMCACHE_CONFIG', '');
