永久存储 kvstore(需开启MongoDB服务)

ecae_kvstore_write()

描述

设置指定的$key为$value, 返回值为bool类型

语法

ecae_kvstore_write($key, $value)

参数 描述
$key string
$value string

例子

<?php
    $key = 'newid';
    $val = time();

    var_dump(ecae_kvstore_write($key, $val));

运行结果

bool(true)

ecae_kvstore_read()

描述

获指定$key的值以$value传递, 返回值为bool类型

语法

ecae_kvstore_read($key, $value)

参数 描述
$key $value
$value &$value

例子

<?php
    $key = 'newid';
    $val = time();
    ecae_kvstore_write($key, $val);

    ecae_kvstore_read($key, $v);
    var_dump($v);

运行结果

int(1319007956)

ecae_kvstore_delete()

描述

删除指定$key, 返回值为bool类型

语法

ecae_kvstore_delete($key)

参数 描述
$key $value

例子

<?php
    $key = 'newid';
    $val = time();
    ecae_kvstore_write($key, $val);
    ecae_kvstore_read($key, $v);

    var_dump(ecae_kvstore_delete($key));

运行结果

bool(true)

ecae_kvstore_list()

描述

语法

ecae_kvstore_list()

例子

<?php
    var_dump(ecae_kvstore_list());
