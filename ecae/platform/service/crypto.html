Crypto

服务概述

ECAE平台提供的加密模块。

查看API rsa

使用指南

<?php
    // 初始化
    ecae_rsa_init();

    $str = "中华人民共和国";

    // 加密
    $str_a = base64_encode(ecae_rsa_encrypt($str));

    // 解密
    $str_b = ecae_rsa_decrypt(base64_decode($str_a));

    echo "</br>加密的字符串：{$str_a}</br>解密的字符串：{$str_b}";
